class Student {
    String name;
    int age;
    int gradeLevel;

    Student (this.name, this.age, this.gradeLevel);

    void printInfo() {
        print('Student: $name, Age: $age, Grade: $gradeLevel');
    }

class Teacher {
    String name;
    int age;
    String subject;

    Teacher (this.name, this.age, this.subject);

    void printInfo() {
        print('Teacher: $name, Age: $age, Subject: $subject');
    }
}

class School {
    void createAndPrintInfo(){
        Student student = new Student('John', 15, 10);
        Teacher teacher = Teacher('Mr.Miya', 23, 'Programming');

        student.printInfo();
        teacher.printInfo();
    }
}

void(){
    School school = new School();
    School createAndPrintInfo();
}
